#include<bits/stdc++.h>
using namespace std;
#define endl '\n';
typedef long long ll;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    cin>>n;
    vector<int>ans;
    while (n!=1) {
        int cst=0;
        if (n&1) {
            cst=1;
            n-=1;
        }
        else {
            cst=n/2;
            n=n-cst;
        }
        ans.push_back(cst);
    }
    int tc=0;
    for (int an : ans) {
        tc+=an;
    }
    cout<<tc<<endl
}
